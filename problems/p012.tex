\begin{problem}{Integer Triangles}
    Two triangles with integer length sides are called \emph{compatible} if:
    \begin{enumerate}
        \item They have the same area,
        \item They have the same perimeter, and
        \item They are not congruent.
    \end{enumerate}

    For example, \((13, 9, 6)\) and \((12, 12, 4)\) are compatible. Find the smallest perimeter admitting at least \(10\) pairwise compatible triangles.
    \end{problem}
